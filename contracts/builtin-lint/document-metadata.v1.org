** Document title
:PROPERTIES:
:ASSERT_ID: document.has-title-keyword
:SEVERITY: warning
:END:
#+begin_src org-contract

  (and
    (keyword :context "metadata")
    (= (summary key) "TITLE")))
#+end_src

#+end_src

#+end_src
